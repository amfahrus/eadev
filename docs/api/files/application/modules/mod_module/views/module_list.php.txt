<script type="text/javascript">
    $(document).ready(function() {
        var lebar = $('.inbody').height() - 200;
        var panjang = $('.content').width() - 20;
        jQuery("#list2").jqGrid({
            url: root + 'mod_module/module_json', 
            mtype : "post",
            datatype: "json", 
            colNames:['id_modules','Module','Icon','Parent','Publish','Link','Sort',''], 
            colModel:[ 
                {name:'id_modules',index:'id_modules',hidden:true, width:100}, 
                {name:'modules',index:'modules', width:200}, 
                {name:'icon',index:'icon', width:150}, 
                {name:'parent',index:'parent', width:150}, 
                {name:'publish',index:'publish', width:150},
                {name:'link',index:'link', width:150},
                {name:'sort',index:'sort', width:150},
                {name:'aksi',width:30, align:"center"}], 
            rowNum:10, 
            width: panjang, 
            height: lebar, 
            rownumbers: true, 
            rownumWidth: 40,
            rowList:[1,10,20,30], 
            pager: '#pager2', 
            multiselect: true,
            viewrecords: true, 
            sortorder: "desc",
            shrinkToFit: false
        }); 
        jQuery("#list2").jqGrid('navGrid','#pager2',{edit:false,add:false,del:false,search:false});
        
        $('#button_search').click(function () {
            var str = $("form").serialize();
            var search = "_search=true&"+ str;
            jQuery("#list2").jqGrid('setGridParam',{
                url: root + 'mod_module/module_json?'+ search,
                page:1
            }).trigger("reloadGrid");
        });
        
        $("#reset_search").click(function() {
            $('.cols_cari').val("");
            $('.ops_cari').val("");
            $('.text').val("");
            jQuery("#list2").jqGrid('setGridParam',{
                url: root + 'mod_module/module_json',
                page:1
            }).trigger("reloadGrid");
        });
        
        $("#form_module_delete").click(function() {
            var id = jQuery("#list2").jqGrid('getGridParam','selarrrow'); 
            if(id){
                $.ajax({
                    url: root + 'mod_module/module_delete',
                    type: 'post',
                    data: { id: id},
                    success: function() {
                        jQuery("#list2").jqGrid().trigger('reloadGrid');
                    }
                });	
            } else {
                alert("Please Select Row to delete!");
            }
        });
    });
</script>
